Для настройки модуля DataBase вначале нужно включить его поддержку в файле bootstrap.php. Для этого нужно раскомментировать строку 'database' в настроечном массиве, который передается методу Kohana::modules:

Kohana::modules(array(

'auth' => MODPATH.'auth', // Basic authentication

// 'cache' => MODPATH.'cache', // Caching with multiple backends

// 'codebench' => MODPATH.'codebench', // Benchmarking tool

'database' => MODPATH.'database', // Database access

// 'image' => MODPATH.'image', // Image manipulation

// 'orm' => MODPATH.'orm', // Object Relationship Mapping

// 'unittest' => MODPATH.'unittest', // Unit testing

// 'userguide' => MODPATH.'userguide', // User guide and API documentation

));

Далее нужно найти файл /SYSTEM/modules/database/config/database.php и скопировать его в каталог /APPLICATION/config.

В этом скопированном файле надо указать значения 4-х переменных: hostname, database, username, password.
