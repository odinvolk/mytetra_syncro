Небольшой пример исключения на языке PHP


function funcDiv($a, $b)

{

if ( $b==0 ) {

throw new \Exception("Cant div by zero\n");

}


return $a/$b;

}



try

{

// Вызов функции. Внутри этой функции мы можем бросить исключение оператором throw

$a = funcDiv(5, 0);

}

catch(Exception $e)

{

echo $e->getMessage();

}
