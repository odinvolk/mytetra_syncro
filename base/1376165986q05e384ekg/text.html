Самый простой вариант:


var title = 'Мой заголовок';

var msg = 'Модель DOM готова...';

Ext.MessageBox.alert(title,msg);


Окно, запрашивающее некоторые данные у пользователя:


function myCallback(btn, text) {

console.info('Вы нажали ' + btn);

if (text) {

console.info('Вы ввели '+ text);

}

}

var title = 'Ввод данных';

var msg = 'Введите какой-нибудь текст.';

Ext.MessageBox.prompt(title,msg,myCallback);


Еще пример через метод show():


var myCallback = function(btn, text) {

console.info('Вы нажали ' + btn);

if (text) {

console.info('Вы ввели '+ text);

}

}


Ext.MessageBox.show({

title : 'Будьте осторожны!',

msg : 'Вы уверены?',

width : 300,

buttons : Ext.MessageBox.YESNOCANCEL,

fn : myCallback,

icon : Ext.MessageBox.QUESTION

});
