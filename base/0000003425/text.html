Чтобы <div> вытеснял собой нужный объем, и не делал перенос после себя, нужно использовать такие CSS свойства:

<html>

<head>

<style type="text/css">

#main {

background-color: green;

width: 250px;

height: 25px;

}

#main .boxy {

background-color: teal;

width: 25px;

height: 25px;

float: left;

}

</style>

</head>

<body>

<div id="main">

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

<div class="boxy">+</div>

</div>

</body>

</html>
